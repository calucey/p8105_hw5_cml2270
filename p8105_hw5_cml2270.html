city_prop_unsolved = function(cityname) {
  city_unsolved =
    n_homs %>% 
    filter(city_state == cityname)
  
  prop_results =
    prop.test(x = city_unsolved %>% pull(n_unsolved), 
              n = city_unsolved %>% pull(n_total_homs), 
              alternative = "two.sided", 
              conf.level = 0.95, 
              correct = FALSE) %>% 
  broom::tidy() %>% 
  select(estimate, conf.low, conf.high) %>% 
    mutate(city_name = cityname) %>% 
    as.tibble()
  return(prop_results)
}

Write a function that takes a vector as an argument; replaces missing values using the rules defined above; and returns the resulting vector. Apply this function to the columns of iris_with_missing using a map statement.

add_missing = function(x){
  
  if (is.double(x)) {
    avg = round(mean(x, na.rm = TRUE), digits = 2)
    x = replace_na(x, avg)
    return(x)
  }
  
  if (is.character(x)) {
    x = replace_na(x, "virginica")
    return(x)
  }
}

replaced_df = map_df(iris_with_missing, add_missing)
